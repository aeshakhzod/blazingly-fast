* Tailwind CSS
#+BEGIN_SRC emacs-lisp
(after! lsp-mode
  (lsp-register-client
   (make-lsp-client
    :new-connection (lsp-stdio-connection '("tailwindcss-language-server" "--stdio"))
    :activation-fn (lsp-activate-on "html" "css" "javascript" "typescript"
                                    "javascriptreact" "typescriptreact" "vue")
    :server-id 'tailwindcss
    :priority -1
    :add      t
    :initialization-options (lambda () (list :userLanguages (h)))
    :notification-handlers (ht ("@/tailwindCSS/projectInitialized" #'ignore)
                               ("@/tailwindCSS/projectsDestroyed" #'ignore)
                               ("tailwindcss/configUpdated" #'ignore)
                               ("tailwindcss/configError" #'ignore)))))
#+END_SRC
